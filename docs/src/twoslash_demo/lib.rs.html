1//! # Twoslash Rustdoc Demo
2//!
3//! This crate demonstrates **twoslash-style type annotations** in rustdoc.
4//! Hover over identifiers in the code examples below to see their inferred types.
5//!
6//! ## Quick Example
7//!
8//! ```rust
9//! use std::collections::HashMap;
10//!
11//! let mut scores = HashMap::new();
12//! scores.insert("Alice", 95);
13//! scores.insert("Bob", 87);
14//!
15//! let total: i32 = scores.values().sum();
16//! let avg = total as f64 / scores.len() as f64;
17//! ```
18//!
19//! ## How This Works
20//!
21//! The type annotations you see on hover are produced by a
22//! [modified version of rustdoc](https://github.com/tmm/rustdoc-twoslash)
23//! that integrates [`twoslash-rust`](https://github.com/wevm/vocs/tree/next/twoslash-rust),
24//! a library that extracts type information from Rust code using rust-analyzer.
25//!
26//! ```text
27//! Doc comment ````rust code blocks
28//!         │
29//!         ▼
30//!    pulldown-cmark (markdown.rs)
31//!         │
32//!         ▼
33//!    CodeBlocks iterator extracts code text
34//!         │
35//!         ▼
36//!    twoslash.rs ──► twoslash-rust ──► rust-analyzer
37//!         │              │                  │
38//!         │              │    scaffolds a temp Cargo project,
39//!         │              │    runs rust-analyzer, returns
40//!         │              │    type info per identifier
41//!         │              ◄──────────────────┘
42//!         ▼
43//!    highlight.rs merges annotations into DecorationInfo
44//!         │
45//!         ▼
46//!    push_token() emits <span data-type="..."> attributes
47//!         │
48//!         ▼
49//!    rustdoc.css renders hover popovers via CSS + JS
50//! ```
51//!
52//! ### Source
53//!
54//! - **Rustdoc fork**: [`tmm/rustdoc-twoslash`](https://github.com/tmm/rustdoc-twoslash) — the modified `src/librustdoc/` files and a patch
55//! - **twoslash-rust**: [`wevm/vocs/twoslash-rust`](https://github.com/wevm/vocs/tree/next/twoslash-rust) — rust-analyzer integration library
56//! - **This demo**: [`tmm/twoslash-demo`](https://github.com/tmm/twoslash-demo)
57//!
58//! ### Build
59//!
60//! ```bash
61//! RUSTDOC_TWOSLASH=1 \
62//!   RUSTDOC=.../build/host/stage1/bin/rustdoc \
63//!   RUSTC=.../build/host/stage1/bin/rustc \
64//!   cargo doc --no-deps
65//! ```

69/// A strongly-typed configuration builder.
70///
71/// # Examples
72///
73/// ```rust
74/// let config = twoslash_demo::Config::new("my-app")
75///     .with_port(8080)
76///     .with_debug(true);
77///
78/// let name = config.name();
79/// let addr = config.address();
80/// ```
81pub struct Config {
82    name: String,
83    port: u16,
84    debug: bool,
85}
86
87impl Config {
88    /// Create a new configuration with the given application name.
89    pub fn new(name: &str) -> Self {
90        Self {
91            name: name.to_string(),
92            port: 3000,
93            debug: false,
94        }
95    }
96
97    /// Set the port number.
98    pub fn with_port(mut self, port: u16) -> Self {
99        self.port = port;
100        self
101    }
102
103    /// Enable or disable debug mode.
104    pub fn with_debug(mut self, debug: bool) -> Self {
105        self.debug = debug;
106        self
107    }
108
109    /// Get the application name.
110    pub fn name(&self) -> &str {
111        &self.name
112    }
113
114    /// Get the full address string.
115    ///
116    /// # Examples
117    ///
118    /// ```rust
119    /// let config = twoslash_demo::Config::new("app").with_port(9090);
120    /// let addr = config.address();
121    /// assert_eq!(addr, "0.0.0.0:9090");
122    /// ```
123    pub fn address(&self) -> String {
124        format!("0.0.0.0:{}", self.port)
125    }
126}

128/// Process a list of items, filtering and transforming them.
129///
130/// # Examples
131///
132/// ```rust
133/// let words = vec!["hello", "world", "rust", "is", "great"];
134/// let result = twoslash_demo::process_items(&words, 3);
135///
136/// let first = result.first();
137/// let count = result.len();
138/// ```
139pub fn process_items(items: &[&str], min_len: usize) -> Vec<String> {
140    items
141        .iter()
142        .filter(|s| s.len() >= min_len)
143        .map(|s| s.to_uppercase())
144        .collect()
145}

147/// A generic key-value store.
148///
149/// # Examples
150///
151/// ```rust
152/// let mut store = twoslash_demo::Store::<String, Vec<i32>>::new();
153/// store.set("scores".into(), vec![100, 95, 87]);
154///
155/// let scores = store.get("scores");
156/// let has_key = store.contains("scores");
157/// ```
158pub struct Store<K, V> {
159    data: HashMap<K, V>,
160}
161
162impl<K: std::hash::Hash + Eq, V> Store<K, V> {
163    /// Create an empty store.
164    pub fn new() -> Self {
165        Self {
166            data: HashMap::new(),
167        }
168    }
169
170    /// Insert a key-value pair.
171    pub fn set(&mut self, key: K, value: V) {
172        self.data.insert(key, value);
173    }
174
175    /// Get a reference to a value by key.
176    pub fn get(&self, key: &K) -> Option<&V> {
177        self.data.get(key)
178    }
179
180    /// Check if a key exists.
181    pub fn contains(&self, key: &K) -> bool {
182        self.data.contains_key(key)
183    }
184}
